inner join users u on u.id=ui.userId

update ignore usersSnapshot uss
inner join usersOrgsInformation ui
on ui.userId=uss.id and ui.orgId=uss.orgId
and ui.disabled=1
inner join users u on u.id=ui.userId
set uss.email=u.email
where ui.disabled=1;
